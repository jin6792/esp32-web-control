  <h1>控制 ESP32 LED</h1>
  <button onclick="controlESP('on')">開燈</button>
  <button onclick="controlESP('off')">關燈</button>

  <script>
    const ESP32_IP = "http://192.168.1.123"; // ← 換成你的 ESP32 IP

    function controlESP(cmd) {
      fetch(`${ESP32_IP}/${cmd}`)
        .then(res => alert(`送出 ${cmd} 成功！`))
        .catch(err => alert("ESP32 沒回應"));
    }
  </script>
